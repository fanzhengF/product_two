	var $aLi = $('.box .item');
	
	//alert
	$aLi.eq(0).on('click', function (){
		var now = 0;
		alertFn();

		function alertFn(){
			layer.alert('点击确定更换图标，当前图标：'+now,{icon:now},function (index){
				//layer.close(index); 	关闭弹窗
				now++;
				if (now==7) {
					layer.alert('自定义样式', {
					    skin: 'layui-layer-molv' //样式类名
					}, function(){
					    layer.alert('深蓝style', {
					        skin: 'layui-layer-lan',
					        shift: 3 //动画类型 1234
					    });
					});
					return;
				};
				alertFn();
			});
		}
	});

	//msg
	$aLi.eq(1).on('click', function (){
		layer.msg('玩命学习中');
	});

	//confirm
	$aLi.eq(2).on('click', function (){
		layer.confirm('你觉得FL怎么样？',{
			btn: 	['萌萌哒','国色天香']
			//,shade: 	false 	遮罩
		},function (){
			layer.msg('很傻很天真',{icon:1});
		},function (){
			layer.msg('漂亮么么哒',{icon:2});
		});
	});

	//open
	$aLi.eq(3).on('click', function (){
		layer.open({
		    type: 1,

		    area: ['420px', '240px'], //宽高
		    content: '我是万恶的弹窗'
		});
	});

	//iframe2
	$aLi.eq(6).on('click', function (){
		layer.open({
		    type: 2,
		    title: false,
		    closeBtn: false,
		    shade: [0],
		    area: ['340px', '215px'],
		    offset: 'rb', //右下角弹出
		    time: 2000, //2秒后自动关闭
		    shift: 1, 	//出来的运动方式 1234
		    content: ['http://jiaju.sina.com.cn/', 'no'], //iframe的url，no代表不显示滚动条
		    end: function(){ //此处用于演示
		        layer.open({
		            type: 2,
		            title: '很多时候，我们想最大化看，比如像这个页面。',
		            shadeClose: true,

	//load
	$aLi.eq(7).on('click', function (){
		var index = layer.load(1, { 	/// 0 1 2 
		    shade: [0.4,'#000'] //0.4透明度的白色背景
		});

		setTimeout(function (){
			layer.close(index)
		},5000);
	});







	//扩展方法
	//prompt、tab层、相册层

	layer.ready(function (){
		//prompt
		$aLi.eq(8).on('click', function (){
			layer.prompt({
				title: '输入任何口令，并确认',
				formType: 1 //prompt风格，支持0-2

				layer.prompt({title: '写爱FL，并确认', formType: 2}, function(text){
				    layer.msg('记录完成！您的口令：'+ pass +' 您最后写下了：'+ text);
				});
			});
		});

		//tab
